大小写转换(lower,upper,initcap)

select lower('Hello World') 转小写，
        upper('hello world') 转大写,
        initcap('hello world') 首字母大写
from dual；

查询结果:

转小写           转大写          首字母大写
---------- -------------------  ---------
hello world        HELLO WORLD     Hello World

子串(substr)

--select substr(a,b,c) 从a字符串中，从第b位开始取，取到第c位

select substr('Hello World',3,4) 子串 
from dual;

查询结果:

子串
-----
llo

字符/字节 数(length/lengthb)

例1）select length('hello world') 字符,
lengthb('hello world') 字节
from dual;

查询结果:

字符     字节
------ ------
 11      11

例2）select length('北京') 字符, lengthb('北京') 字节 
from dual;

查询结果: 

字符     字节
------ ------
 2        4

字串位置查询(instr(a,b))

select instr('hello world','ll') ll的位置
from dual;

查询结果:

LL的位置
-----
 3

左右填充(lpad(‘?’,n,’?’) rpad(‘?’,n,’?’))

说明: 10为补充至10个字节
select lpad('abcd',10,'*') 左,
rpad('abcd',10,'*') 右 
from dual;

查询结果:

左         右 
---------- ----------    
******abcd abcd******

round 四舍五入

select round(45.926,2) 一,round(45.926,1) 二,
round(45.926,0) 三,round(45.926,-1) 四,
round(45.926,-2) 五
 from dual;         

查询结果:

 一         二         三         四         五 
---------- ---------- ---------- ------- --------
 45.93       45.9         46         50      0

trunc 截断

select trunc(45.926,2) 一,trunc(45.926,1) 二,
trunc(45.926,0) 三,
trunc(45.926,-1) 四,trunc(45.926,-2) 五
from dual

查询结果:

  一         二         三         四         五 
------ ---------- ---------- ---------- -------------
 45.92       45.9         45         40       0

(sysdate-hiredate)/365 年

nvl2(a,b,c) 当a=null的时候，返回c；否则返回b

select sal*12+nvl2(comm,comm,0) from emp;

查询结果:

SAL*12+NVL2(COMM,COMM,0)   
------------------------
    9600
   19500
   15500
   35700

nullif(a,b) 当a=b的时候，返回null；否则返回a

例1:
select nullif('abc','abc') 值 from dual;
查询结果:
值
---

例2:
select nullif('abc','abcd') 值 from dual;
查询结果:
值
---
abc

coalesce 从左到右 找到第一个不为null的值

select comm,sal,coalesce(comm,sal) 
"第一个不为null的值" 
from emp;

查询结果:

      COMM   SAL 第一个不为null的值  
----------- ----- ------------------  
             800                800 
   300      1600                300
   500      1250                500  
                2975               2975
   1400      1250               1400

涨工资，总裁1000 经理800 其他400

方法一:
select empno,ename,job,sal 涨前,
       case job when 'PRESIDENT' then sal+1000
                when 'MANAGER' then sal+800
                else sal+400
        end 涨后
from emp;   

方法二:
select empno,ename,job,sal 涨前,
       decode(job,'PRESIDENT',sal+1000,
                  'MANAGER',sal+800,
                          sal+400) 涨后
from emp;

查询结果:

  EMPNO ENAME    JOB             涨前       涨后    
  ---------- -------- --------- ---------- ----------      
  7499 ALLEN    SALESMAN        1600       2000
  7521 WARD     SALESMAN        1250       1650
  7566 JONES    MANAGER         2975       3775 
  7654 MARTIN   SALESMAN        1250       1650
  7698 BLAKE    MANAGER         2850       3650  
  7782 CLARK    MANAGER         2450       3250
